-- Questão 1
SELECT 
    A.MAT_ALU AS MATRICULAS, A.NOM_ALU as NOMES, A.DAT_NASC AS Data_Nascimento, A.MGP
FROM
    ALUNOS AS A ORDER BY NOMES ASC
-- Questão 2
SELECT 
    C.TOT_CRED, C.NOM_CURSO
FROM
    CURSOS AS C
ORDER BY TOT_CRED DESC;
SELECT 
    C.NOM_CURSO, C.TOT_CRED
FROM
    CURSOS AS C
ORDER BY NOM_CURSO ASC;
-- Questão 3
SELECT 
    P.NOM_PROF
FROM
    PROFESSORES AS P
WHERE
    P.NOM_PROF LIKE 'MARIA%'
ORDER BY P.NOM_PROF ASC;
-- Questão 4
SELECT
	A.MAT_ALU, A.NOM_ALU AS NOMES, A.MGP
FROM
	ALUNOS AS A
WHERE
	A.MGP > 8 
ORDER BY A.MGP DESC;
-- Questão 5
SELECT
	COD_DISC, D.NOM_DISC AS "Nome Disciplina", D.CREDITOS
FROM
	DISCIPLINAS AS D
WHERE
	D.CREDITOS > 4
ORDER BY D.NOM_DISC ASC;
-- Questão 6
SELECT 
    A.MAT_ALU, A.NOM_ALU AS NOMES, A.DAT_NASC
FROM
    ALUNOS AS A
WHERE
    A.DAT_NASC > '2000-12-31'
ORDER BY A.NOM_ALU ASC
-- QUESTÃO 7
SELECT 
    A.NOM_ALU AS NOMES, C.NOM_CURSO AS CURSOS
FROM
    ALUNOS AS A,
    CURSOS AS C
WHERE
    A.COD_CURSO = C.COD_CURSO
ORDER BY A.NOM_ALU ASC
